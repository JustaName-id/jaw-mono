# account.getPermission()

Get details of a previously granted permission.

**Type:** `instance async`

---

## Signature

```typescript
async getPermission(permissionId: Hex): Promise<WalletGrantPermissionsResponse>
```

---

## Parameters

### permissionId

Type: `Hex`

The permission ID (hash) to fetch. This is the `id` field returned from `grantPermissions()`.

---

## Returns

`Promise<WalletGrantPermissionsResponse>` - The permission details.

```typescript
interface WalletGrantPermissionsResponse {
  /** Smart account address */
  address: Address;
  /** Chain ID as hex string */
  chainId: Hex;
  /** Start timestamp */
  start: number;
  /** Expiry timestamp */
  expiry: number;
  /** Unique salt for this permission */
  salt: Hex;
  /** Permission ID (hash) */
  id: Hex;
  /** Spender address */
  spender: Address;
  /** Granted call permissions */
  calls: CallPermissionDetail[];
  /** Granted spend permissions */
  spends: SpendPermissionDetail[];
}
```

---

## Behavior

1. Fetches the permission from the JAW relay by ID
2. Returns the full permission details in the same format as `grantPermissions()`
3. Throws if the permission is not found

---

## Examples

### Basic Usage

```typescript
import { Account } from '@jaw.id/core';

const account = await Account.get({
  chainId: 1,
  apiKey: 'your-api-key',
});

const permission = await account.getPermission(
  '0x1234567890abcdef...' // Permission ID
);

console.log('Spender:', permission.spender);
console.log('Expires:', new Date(permission.expiry * 1000));
console.log('Call permissions:', permission.calls);
console.log('Spend permissions:', permission.spends);
```

### Check Permission Status

```typescript
async function isPermissionValid(
  account: Account,
  permissionId: Hex
): Promise<boolean> {
  try {
    const permission = await account.getPermission(permissionId);

    // Check if expired
    const now = Math.floor(Date.now() / 1000);
    if (permission.expiry < now) {
      console.log('Permission has expired');
      return false;
    }

    // Check if not yet started
    if (permission.start > now) {
      console.log('Permission not yet active');
      return false;
    }

    return true;
  } catch (error) {
    console.log('Permission not found or revoked');
    return false;
  }
}
```

### Display Permission Details

```typescript
function PermissionDetails({ account, permissionId }: Props) {
  const [permission, setPermission] = useState<WalletGrantPermissionsResponse | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    account.getPermission(permissionId)
      .then(setPermission)
      .catch(console.error)
      .finally(() => setLoading(false));
  }, [permissionId]);

  if (loading) return <div>Loading...</div>;
  if (!permission) return <div>Permission not found</div>;

  return (
    <div>
      <h3>Permission Details</h3>
      <p>Spender: {permission.spender}</p>
      <p>Expires: {new Date(permission.expiry * 1000).toLocaleString()}</p>

      {permission.calls.length > 0 && (
        <div>
          <h4>Call Permissions</h4>
          {permission.calls.map((call, i) => (
            <p key={i}>
              {call.target} - {call.selector}
            </p>
          ))}
        </div>
      )}

      {permission.spends.length > 0 && (
        <div>
          <h4>Spend Permissions</h4>
          {permission.spends.map((spend, i) => (
            <p key={i}>
              {spend.token}: {spend.limit} per {spend.period}
            </p>
          ))}
        </div>
      )}
    </div>
  );
}
```

### Verify Permission After Grant

```typescript
// Grant permission
const grantResult = await account.grantPermissions(
  Math.floor(Date.now() / 1000) + 3600,
  '0xSpender...',
  { calls: [{ target: '0x...', selector: '0x...' }] }
);

// Verify it was stored correctly
const fetched = await account.getPermission(grantResult.id);

// Should match
console.log('Grant result:', grantResult);
console.log('Fetched:', fetched);
console.log('Match:', grantResult.id === fetched.id);
```

### List All Active Permissions

```typescript
// Store permission IDs when granting
const grantedPermissions: Hex[] = [];

// Grant a permission
const permission = await account.grantPermissions(/* ... */);
grantedPermissions.push(permission.id);

// Later, fetch all active permissions
async function getActivePermissions(
  account: Account,
  permissionIds: Hex[]
) {
  const now = Math.floor(Date.now() / 1000);
  const permissions = await Promise.all(
    permissionIds.map(id =>
      account.getPermission(id).catch(() => null)
    )
  );

  return permissions.filter(
    p => p !== null && p.expiry > now
  );
}
```

## Error Handling

```typescript
try {
  const permission = await account.getPermission(permissionId);
} catch (error) {
  if (error.message.includes('not found')) {
    console.log('Permission does not exist or was revoked');
  } else {
    console.error('Failed to fetch permission:', error);
  }
}
```
