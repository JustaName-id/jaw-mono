# account.getMetadata()

Get account metadata (only available for passkey-based accounts).

**Type:** `instance`

---

## Signature

```typescript
getMetadata(): AccountMetadata | null
```

---

## Parameters

None.

---

## Returns

`AccountMetadata | null` - Account metadata or `null` for LocalAccount-based accounts.

### AccountMetadata

```typescript
interface AccountMetadata {
  /** Username/display name */
  username: string;
  /** ISO date string when the account was created */
  creationDate: string;
  /** Whether the account was imported from cloud backup */
  isImported: boolean;
}
```

---

## Behavior

- For accounts created via `Account.create()`: Returns full metadata
- For accounts created via `Account.import()`: Returns metadata with `isImported: true`
- For accounts created via `Account.fromLocalAccount()`: Returns `null`

---

## Examples

### Basic Usage

```typescript
import { Account } from '@jaw.id/core';

const account = await Account.get({
  chainId: 1,
  apiKey: 'your-api-key',
});

const metadata = account.getMetadata();

if (metadata) {
  console.log('Username:', metadata.username);
  console.log('Created:', metadata.creationDate);
  console.log('Imported:', metadata.isImported);
} else {
  console.log('Local account - no metadata available');
}
```

### Display Account Info

```typescript
import { Account } from '@jaw.id/core';

function AccountInfo({ account }: { account: Account }) {
  const metadata = account.getMetadata();

  return (
    <div>
      <p>Address: {account.address}</p>
      {metadata ? (
        <>
          <p>Username: {metadata.username}</p>
          <p>
            Created: {new Date(metadata.creationDate).toLocaleDateString()}
          </p>
          {metadata.isImported && <span>Imported from cloud</span>}
        </>
      ) : (
        <p>Local account</p>
      )}
    </div>
  );
}
```

### Conditional Features

```typescript
import { Account } from '@jaw.id/core';

async function setupAccount(account: Account) {
  const metadata = account.getMetadata();

  if (metadata) {
    // Passkey account - can use cloud sync features
    enableCloudSync();
  } else {
    // Local account - show backup warning
    showBackupWarning();
  }
}
```

### Format Creation Date

```typescript
const metadata = account.getMetadata();

if (metadata) {
  const createdAt = new Date(metadata.creationDate);

  // Relative time
  const rtf = new Intl.RelativeTimeFormat('en', { numeric: 'auto' });
  const daysAgo = Math.floor(
    (Date.now() - createdAt.getTime()) / (1000 * 60 * 60 * 24)
  );
  console.log(`Created ${rtf.format(-daysAgo, 'day')}`);

  // Formatted date
  console.log(createdAt.toLocaleDateString('en-US', {
    year: 'numeric',
    month: 'long',
    day: 'numeric',
  }));
}
```
